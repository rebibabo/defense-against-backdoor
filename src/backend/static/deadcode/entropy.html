def normal(ken, nao):
    total = 0
    k = ken[:]
    for i in nao[::-1]:
        if i > ken[-1]:
            k = k[1:]
        else:
            for e in range(len(k)):
                if k[e] > i:
                    del k[e]
                    total += 1
                    break
    return total
